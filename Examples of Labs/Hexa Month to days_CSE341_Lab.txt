INCLUDE "EMU8086.INC"
.MODEL SMALL
.STACK 100H

.DATA
BORO DB "31 days$"
CHOTO DB "30 days$"
FEB DB "28 days$"
NOTICE DB "INVALID MONTH!$"
.CODE
MAIN PROC

;iniitialize DS

MOV AX,@DATA
MOV DS,AX
 
; enter your code here

MOV AH, 1
INT 21H 
PRINTN
MOV BL,AL
CMP BL, 043H
JG PROPER
CMP BL, 031H
JL PROPER
CMP BL, 040H
JG ATLAST

CMP BL, 038H
JL PUNY 
MOV AH,0
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG  C
JMP FINAL

PUNY:
MOV AH,0
CMP BL, 032H
JE LAST
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG FINAL

C:
LEA DX,CHOTO
MOV AH, 9  
INT 21H
JMP EXIT


FINAL:   
LEA DX,BORO
MOV AH, 9  
INT 21H
JMP EXIT

LAST:
LEA DX,FEB
MOV AH, 9  
INT 21H
JMP EXIT  

ATLAST:
MOV AH,0
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG  FINAL
JMP C

PROPER:
LEA DX,NOTICE
MOV AH, 9  
INT 21H
JMP EXIT
EXIT:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
