.MODEL SMALL

.DATA
NUM DB 48  
MSG1 DB " WHICH IS AN EVEN NUMBER$"
MSG2 DB " WHICH IS A ODD NUMBER$"

.CODE
MAIN PROC

;iniitialize DS
MOV AX,@DATA
MOV DS,AX
 
; enter your code here

MOV AH,1
INT 21H
MOV BH, AL

MOV AH,1
INT 21H
MOV CH, AL 

MOV AH,1
INT 21H
MOV CL, AL  

CMP BH, CH
JG BHb      

CMP CH, CL  
JG CHb
MOV DL, CL 
PRINTN
MOV AH, 2
INT 21H
JMP EXIT

BHb:
CMP BH, CL
JG FINAL
MOV DL, CL  
PRINTN
MOV AH, 2
INT 21H
JMP EXIT

CHb:
MOV DL,CH 
PRINTN
MOV AH,2
INT 21H
JMP EXIT

FINAL:
MOV DL,BH 
PRINTN
MOV AH,2
INT 21H
JMP EXIT

;exit to DOS
EXIT:
MOV BL,2   
MOV AH, 0
MOV AL, DL
DIV BL

CMP AH, 0D
JG SAD

LEA DX,MSG1
MOV AH, 9
INT 21H
JMP PROPER

SAD:
LEA DX,MSG2
MOV AH, 9
INT 21H 
JMP PROPER

PROPER:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
