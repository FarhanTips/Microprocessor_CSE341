INCLUDE "EMU8086.INC"
.MODEL SMALL
.STACK 100H

.DATA 
MSG1 DB "  PLEASE FOLLOW THE INSTRUCTIONS!!!!$"
MSG2 DB "  01 = JANUARY, 1 = INVALID MONTH, 10 = OCTOBER$"
MSG3 DB "  12 = DECEMBER, 14 = INVALID MONTH$"
MSG4 DB "PLEASE TYPE MONTH NUMBER ACCORDING TO THE INSTRUCTIONS: $"    
MSG5 DB "OUTPUT: $"
BORO DB "31 days$"
CHOTO DB "30 days$"
FEB DB "28 days$"
NOTICE DB "INVALID MONTH! PLEASE TRY TO FOLLOW THE INSTRUCTIONS WHILE TYPING INPUT$"
.CODE
MAIN PROC

;iniitialize DS

MOV AX,@DATA
MOV DS,AX
 
; enter your code here 
LEA DX, MSG1
MOV AH, 9
INT 21H 

PRINTN 

LEA DX, MSG2
MOV AH, 9
INT 21H
 
PRINTN

LEA DX, MSG3
MOV AH, 9
INT 21H

PRINTN
PRINTN
LEA DX, MSG4
MOV AH, 9
INT 21H

MOV AH, 1
INT 21H 
MOV BH,AL

MOV AH, 1
INT 21H 
MOV BL,AL
PRINTN
LEA DX, MSG5
MOV AH, 9
INT 21H

CMP BH, 030H
JE NEW_DRAMA 

CMP BH, 031H
JE ATLAST  

CMP BH, 031H
JG PROPER



NEW_DRAMA:
CMP BL, 038H
JL PUNY 
MOV AH,0
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG  C
JMP FINAL

PUNY:
MOV AH,0
CMP BL, 032H
JE LAST
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG FINAL
JMP C

C:
LEA DX,CHOTO
MOV AH, 9  
INT 21H
JMP EXIT


FINAL:   
LEA DX,BORO
MOV AH, 9  
INT 21H
JMP EXIT

LAST:
LEA DX,FEB
MOV AH, 9  
INT 21H
JMP EXIT  

ATLAST:
CMP BL,032H
JG PROPER
MOV AH,0
MOV AL, BL
MOV CL,2
DIV CL
CMP AH, 0
JG  C
JMP FINAL

PROPER:
LEA DX,NOTICE
MOV AH, 9  
INT 21H
JMP EXIT  

EXIT:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
