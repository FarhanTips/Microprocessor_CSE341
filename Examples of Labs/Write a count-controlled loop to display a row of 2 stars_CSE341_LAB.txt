

INCLUDE "EMU8086.INC"
.MODEL SMALL
 
.STACK 100H

.DATA
a db 1,2,3,4,5
b db 5 dup(?)
.CODE
MAIN PROC

;iniitialize DS

MOV AX,@DATA
MOV DS,AX
 
; enter your code here 
MOV BH, 48D; BH=0
MOV AH, 1
INT 21H
MOV BL, AL

START2:
INC BH
CMP BH, BL
JG EXIT

MOV CL, 2D
MOV CH, 0D
PRINTN

START:
MOV DL, 02AH
MOV AH,2
INT 21H
INC CH
CMP CH, CL
JGE START2
JMP START

;exit to DOS
EXIT:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
