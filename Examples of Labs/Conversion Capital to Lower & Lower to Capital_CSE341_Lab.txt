INCLUDE "EMU8086.INC"
.MODEL SMALL
 
.STACK 100H

.DATA
a db 1,2,3,4,5
MSG2 DB "YOU GAVE A VALUSE WHICH IS NOT BETWEEN A-Z OR a-z.$"
.CODE
MAIN PROC

;iniitialize DS

MOV AX,@DATA
MOV DS,AX
 
; enter your code here 

MOV AH, 1
INT 21H
MOV BL, AL
PRINTN
CMP BL, 060H
JG BORO 
CMP BL, 041H
JL PROPER
MOV DL,BL
ADD DL, 020H
JMP FINAL

BORO: 
CMP BL, 07AH
JG PROPER
MOV DL,BL
SUB DL, 020H
JMP FINAL

FINAL:
MOV AH, 2
INT 21H
JMP EXIT      

PROPER:
LEA DX, MSG2
MOV AH, 9
INT 21H
JMP EXIT

;exit to DOS
EXIT:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
