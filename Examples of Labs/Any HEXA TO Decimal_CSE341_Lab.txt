;ANY ONE BIT-HEXA TO DECIMAL NUMBER

INCLUDE "EMU8086.INC"
.MODEL SMALL

.DATA  
MSG1 DB "GIVE A NUMBER IN HEXA: $"
MSG2 DB "EQIVALENT DECIMAL NUMBER: $"

.CODE
MAIN PROC

;iniitialize DS
MOV AX,@DATA
MOV DS,AX
 
; enter your code here
LEA DX,MSG1
MOV AH,9
INT 21H

MOV AH,1
INT 21H 
MOV BL, AL 
CMP BL,040H
JG STEP1 

PRINTN
LEA DX,MSG2
MOV AH,9
INT 21H  

MOV DL, BL  
MOV AH, 2
INT 21H
JMP EXIT

STEP1:  
PRINTN
LEA DX,MSG2
MOV AH,9
INT 21H
 
MOV DL,031H  
MOV AH, 2
INT 21H

SUB BL, 17D  
MOV DL, BL
MOV AH, 2
INT 21H

;exit to DOS
EXIT:               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
